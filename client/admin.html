async function loadOrders(){ const res = await fetch("/api/orders"); const orders = await res.json(); let html=""; orders.reverse().forEach(o=>{ html+=`
Order: ${o.orderId}
Player: ${o.uid} (${o.server})
Package: ${o.package}
Price: ₹${o.price}
Status: ${o.status}
`; }); document.getElementById("orders").innerHTML = html; } loadOrders(); setInterval(loadOrders,5000);
